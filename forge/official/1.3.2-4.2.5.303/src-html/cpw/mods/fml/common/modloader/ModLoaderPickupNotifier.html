001    package cpw.mods.fml.common.modloader;
002    
003    import net.minecraft.src.EntityItem;
004    import net.minecraft.src.EntityPlayer;
005    import cpw.mods.fml.common.IPickupNotifier;
006    
007    public class ModLoaderPickupNotifier implements IPickupNotifier
008    {
009    
010        private BaseModProxy mod;
011    
012        public ModLoaderPickupNotifier(BaseModProxy mod)
013        {
014            this.mod = mod;
015        }
016    
017        @Override
018        public void notifyPickup(EntityItem item, EntityPlayer player)
019        {
020            mod.onItemPickup(player, item.item);
021        }
022    
023    }
