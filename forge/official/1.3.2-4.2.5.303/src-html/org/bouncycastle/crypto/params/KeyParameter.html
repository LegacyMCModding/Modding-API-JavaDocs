001    package org.bouncycastle.crypto.params;
002    
003    import org.bouncycastle.crypto.CipherParameters;
004    
005    public class KeyParameter implements CipherParameters
006    {
007        private byte[] key;
008    
009        public KeyParameter(byte[] par1ArrayOfByte)
010        {
011            this(par1ArrayOfByte, 0, par1ArrayOfByte.length);
012        }
013    
014        public KeyParameter(byte[] par1ArrayOfByte, int par2, int par3)
015        {
016            this.key = new byte[par3];
017            System.arraycopy(par1ArrayOfByte, par2, this.key, 0, par3);
018        }
019    
020        public byte[] getKey()
021        {
022            return this.key;
023        }
024    }
