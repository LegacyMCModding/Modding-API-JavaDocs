001    package net.minecraftforge.event.entity;
002    
003    import net.minecraft.src.Entity;
004    import net.minecraftforge.event.Cancelable;
005    
006    @Cancelable
007    public class PlaySoundAtEntityEvent extends EntityEvent
008    { 
009        public String name;
010        public final float volume;
011        public final float pitch;
012        
013        public PlaySoundAtEntityEvent(Entity entity, String name, float volume, float pitch)
014        {
015            super(entity);
016            this.name = name;
017            this.volume = volume;
018            this.pitch = pitch;
019        }
020    }
