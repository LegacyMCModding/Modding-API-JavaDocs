001    package net.minecraftforge.client.event.sound;
002    
003    import net.minecraft.client.audio.SoundManager;
004    
005    /**
006     * This event is raised by the SoundManager when it does its first setup of the 
007     * SoundSystemConfig's codecs, use this function to add your own codecs.
008     */
009    public class SoundSetupEvent extends SoundEvent
010    {
011        public final SoundManager manager;
012        public SoundSetupEvent(SoundManager manager)
013        {
014            this.manager = manager;
015        }
016    }
