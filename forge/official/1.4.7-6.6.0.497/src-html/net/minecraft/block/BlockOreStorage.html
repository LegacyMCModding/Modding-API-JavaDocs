001    package net.minecraft.block;
002    
003    import net.minecraft.block.material.Material;
004    import net.minecraft.creativetab.CreativeTabs;
005    
006    public class BlockOreStorage extends Block
007    {
008        public BlockOreStorage(int par1, int par2)
009        {
010            super(par1, Material.iron);
011            this.blockIndexInTexture = par2;
012            this.setCreativeTab(CreativeTabs.tabBlock);
013        }
014    
015        /**
016         * Returns the block texture based on the side being looked at.  Args: side
017         */
018        public int getBlockTextureFromSide(int par1)
019        {
020            return this.blockIndexInTexture;
021        }
022    }
