001    package net.minecraft.src;
002    
003    import cpw.mods.fml.common.Side;
004    import cpw.mods.fml.common.asm.SideOnly;
005    
006    @SideOnly(Side.CLIENT)
007    public class ChatLine
008    {
009        /** GUI Update Counter value this Line was created at */
010        private final int updateCounterCreated;
011        private final String lineString;
012    
013        /**
014         * int value to refer to existing Chat Lines, can be 0 which means unreferrable
015         */
016        private final int chatLineID;
017    
018        public ChatLine(int par1, String par2Str, int par3)
019        {
020            this.lineString = par2Str;
021            this.updateCounterCreated = par1;
022            this.chatLineID = par3;
023        }
024    
025        public String getChatLineString()
026        {
027            return this.lineString;
028        }
029    
030        public int getUpdatedCounter()
031        {
032            return this.updateCounterCreated;
033        }
034    
035        public int getChatLineID()
036        {
037            return this.chatLineID;
038        }
039    }
