020        protected void renderMooshroomEquippedItems(EntityMooshroom par1EntityMooshroom, float par2)
021        {
022            super.renderEquippedItems(par1EntityMooshroom, par2);
023    
024            if (!par1EntityMooshroom.isChild())
025            {
026                this.loadTexture("/terrain.png");
027                GL11.glEnable(GL11.GL_CULL_FACE);
028                GL11.glPushMatrix();
029                GL11.glScalef(1.0F, -1.0F, 1.0F);
030                GL11.glTranslatef(0.2F, 0.4F, 0.5F);
031                GL11.glRotatef(42.0F, 0.0F, 1.0F, 0.0F);
032                this.renderBlocks.renderBlockAsItem(Block.mushroomRed, 0, 1.0F);
033                GL11.glTranslatef(0.1F, 0.0F, -0.6F);
034                GL11.glRotatef(42.0F, 0.0F, 1.0F, 0.0F);
035                this.renderBlocks.renderBlockAsItem(Block.mushroomRed, 0, 1.0F);
036                GL11.glPopMatrix();
037                GL11.glPushMatrix();
038                ((ModelQuadruped)this.mainModel).head.postRender(0.0625F);
039                GL11.glScalef(1.0F, -1.0F, 1.0F);
040                GL11.glTranslatef(0.0F, 0.75F, -0.2F);
041                GL11.glRotatef(12.0F, 0.0F, 1.0F, 0.0F);
042                this.renderBlocks.renderBlockAsItem(Block.mushroomRed, 0, 1.0F);
043                GL11.glPopMatrix();
044                GL11.glDisable(GL11.GL_CULL_FACE);
045            }
046        }
