001    package net.minecraftforge.event.entity.living;
002    
003    import net.minecraft.src.EntityLiving;
004    import net.minecraftforge.event.Cancelable;
005    import net.minecraftforge.event.entity.EntityEvent;
006    
007    public class LivingEvent extends EntityEvent
008    {
009        public final EntityLiving entityLiving;
010        public LivingEvent(EntityLiving entity)
011        {
012            super(entity);
013            entityLiving = entity;
014        }
015        
016        @Cancelable
017        public static class LivingUpdateEvent extends LivingEvent
018        {
019            public LivingUpdateEvent(EntityLiving e){ super(e); }
020        }
021    
022        public static class LivingJumpEvent extends LivingEvent
023        {
024            public LivingJumpEvent(EntityLiving e){ super(e); }
025        }
026    }
