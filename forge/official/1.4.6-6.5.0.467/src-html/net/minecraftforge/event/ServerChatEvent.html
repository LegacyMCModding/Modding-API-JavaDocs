001    package net.minecraftforge.event;
002    
003    import net.minecraft.entity.player.EntityPlayerMP;
004    
005    @Cancelable
006    public class ServerChatEvent extends Event {
007        public final String message, username;
008        public final EntityPlayerMP player;
009        public String line;
010        public ServerChatEvent(EntityPlayerMP player, String message, String line)
011        {
012            super();
013            this.message = message;
014            this.player = player;
015            this.username = player.username;
016            this.line = "<" + username + "> " + message;
017        }
018    }
