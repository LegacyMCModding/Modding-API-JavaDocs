020         * Generates a random enchant name.
021         */
022        public String generateRandomEnchantName()
023        {
024            int var1 = this.rand.nextInt(2) + 3;
025            String var2 = "";
026    
027            for (int var3 = 0; var3 < var1; ++var3)
028            {
029                if (var3 > 0)
030                {
031                    var2 = var2 + " ";
032                }
033    
034                var2 = var2 + this.wordList[this.rand.nextInt(this.wordList.length)];
035            }
036    
037            return var2;
038        }
039

041         * Sets the seed for the enchant name RNG.
