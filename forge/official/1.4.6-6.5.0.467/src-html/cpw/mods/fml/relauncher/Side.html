015    package cpw.mods.fml.relauncher;
016    
017    public enum Side {
018        CLIENT, SERVER, BUKKIT;
019    
020        /**
021         * @return If this is the server environment
022         */
023        public boolean isServer()
024        {
025            return !isClient();
026        }
027    
028        /**
029         * @return if this is the Client environment
030         */
031        public boolean isClient()
032        {
033            return this == CLIENT;
034        }
035    }
