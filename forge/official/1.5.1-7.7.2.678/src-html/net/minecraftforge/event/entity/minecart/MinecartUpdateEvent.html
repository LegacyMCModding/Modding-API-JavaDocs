001package net.minecraftforge.event.entity.minecart;
002
003import net.minecraft.entity.item.EntityMinecart;
004
005public class MinecartUpdateEvent extends MinecartEvent
006{
007    public final float x;
008    public final float y;
009    public final float z;
010
011    public MinecartUpdateEvent(EntityMinecart minecart, float x, float y, float z)
012    {
013        super(minecart);
014        this.x = x;
015        this.y = y;
016        this.z = z;
017    }
018}
