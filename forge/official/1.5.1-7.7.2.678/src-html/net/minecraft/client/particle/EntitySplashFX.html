001package net.minecraft.client.particle;
002
003import cpw.mods.fml.relauncher.Side;
004import cpw.mods.fml.relauncher.SideOnly;
005import net.minecraft.world.World;
006
007@SideOnly(Side.CLIENT)
008public class EntitySplashFX extends EntityRainFX
009{
010    public EntitySplashFX(World par1World, double par2, double par4, double par6, double par8, double par10, double par12)
011    {
012        super(par1World, par2, par4, par6);
013        this.particleGravity = 0.04F;
014        this.nextTextureIndexX();
015
016        if (par10 == 0.0D && (par8 != 0.0D || par12 != 0.0D))
017        {
018            this.motionX = par8;
019            this.motionY = par10 + 0.1D;
020            this.motionZ = par12;
021        }
022    }
023}
