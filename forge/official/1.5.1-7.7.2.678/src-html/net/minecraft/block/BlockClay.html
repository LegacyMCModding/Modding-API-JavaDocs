001package net.minecraft.block;
002
003import java.util.Random;
004import net.minecraft.block.material.Material;
005import net.minecraft.creativetab.CreativeTabs;
006import net.minecraft.item.Item;
007
008public class BlockClay extends Block
009{
010    public BlockClay(int par1)
011    {
012        super(par1, Material.clay);
013        this.setCreativeTab(CreativeTabs.tabBlock);
014    }
015
016    /**
017     * Returns the ID of the items to drop on destruction.
018     */
019    public int idDropped(int par1, Random par2Random, int par3)
020    {
021        return Item.clay.itemID;
022    }
023
024    /**
025     * Returns the quantity of items to drop on block destruction.
026     */
027    public int quantityDropped(Random par1Random)
028    {
029        return 4;
030    }
031}
