001package net.minecraft.world.biome;
002
003import java.util.Random;
004import net.minecraft.entity.passive.EntityWolf;
005import net.minecraft.world.gen.feature.WorldGenTaiga1;
006import net.minecraft.world.gen.feature.WorldGenTaiga2;
007import net.minecraft.world.gen.feature.WorldGenerator;
008
009public class BiomeGenTaiga extends BiomeGenBase
010{
011    public BiomeGenTaiga(int par1)
012    {
013        super(par1);
014        this.spawnableCreatureList.add(new SpawnListEntry(EntityWolf.class, 8, 4, 4));
015        this.theBiomeDecorator.treesPerChunk = 10;
016        this.theBiomeDecorator.grassPerChunk = 1;
017    }
018
019    /**
020     * Gets a WorldGen appropriate for this biome.
021     */
022    public WorldGenerator getRandomWorldGenForTrees(Random par1Random)
023    {
024        return (WorldGenerator)(par1Random.nextInt(3) == 0 ? new WorldGenTaiga1() : new WorldGenTaiga2(false));
025    }
026}
