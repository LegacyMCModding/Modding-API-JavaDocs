001package net.minecraft.item.crafting;
002
003import net.minecraft.block.Block;
004import net.minecraft.item.Item;
005import net.minecraft.item.ItemStack;
006
007public class RecipesFood
008{
009    /**
010     * Adds the food recipes to the CraftingManager.
011     */
012    public void addRecipes(CraftingManager par1CraftingManager)
013    {
014        par1CraftingManager.addShapelessRecipe(new ItemStack(Item.bowlSoup), new Object[] {Block.mushroomBrown, Block.mushroomRed, Item.bowlEmpty});
015        par1CraftingManager.addRecipe(new ItemStack(Item.cookie, 8), new Object[] {"#X#", 'X', new ItemStack(Item.dyePowder, 1, 3), '#', Item.wheat});
016        par1CraftingManager.addRecipe(new ItemStack(Block.melon), new Object[] {"MMM", "MMM", "MMM", 'M', Item.melon});
017        par1CraftingManager.addRecipe(new ItemStack(Item.melonSeeds), new Object[] {"M", 'M', Item.melon});
018        par1CraftingManager.addRecipe(new ItemStack(Item.pumpkinSeeds, 4), new Object[] {"M", 'M', Block.pumpkin});
019        par1CraftingManager.addShapelessRecipe(new ItemStack(Item.pumpkinPie), new Object[] {Block.pumpkin, Item.sugar, Item.egg});
020        par1CraftingManager.addShapelessRecipe(new ItemStack(Item.fermentedSpiderEye), new Object[] {Item.spiderEye, Block.mushroomBrown, Item.sugar});
021        par1CraftingManager.addShapelessRecipe(new ItemStack(Item.speckledMelon), new Object[] {Item.melon, Item.goldNugget});
022        par1CraftingManager.addShapelessRecipe(new ItemStack(Item.blazePowder, 2), new Object[] {Item.blazeRod});
023        par1CraftingManager.addShapelessRecipe(new ItemStack(Item.magmaCream), new Object[] {Item.blazePowder, Item.slimeBall});
024    }
025}
