248    {
249        this.itemID = 256 + par1;
250
251        if (itemsList[256 + par1] != null)
252        {
253            System.out.println("CONFLICT @ " + par1 + " item slot already occupied by " + itemsList[256 + par1] + " while adding " + this);
254        }
255
256        itemsList[256 + par1] = this;
257
258        GameData.newItemAdded(this);
259    }

629    protected MovingObjectPosition getMovingObjectPositionFromPlayer(World par1World, EntityPlayer par2EntityPlayer, boolean par3)
630    {
631        float f = 1.0F;
632        float f1 = par2EntityPlayer.prevRotationPitch + (par2EntityPlayer.rotationPitch - par2EntityPlayer.prevRotationPitch) * f;
633        float f2 = par2EntityPlayer.prevRotationYaw + (par2EntityPlayer.rotationYaw - par2EntityPlayer.prevRotationYaw) * f;
634        double d0 = par2EntityPlayer.prevPosX + (par2EntityPlayer.posX - par2EntityPlayer.prevPosX) * (double)f;
635        double d1 = par2EntityPlayer.prevPosY + (par2EntityPlayer.posY - par2EntityPlayer.prevPosY) * (double)f + 1.62D - (double)par2EntityPlayer.yOffset;
636        double d2 = par2EntityPlayer.prevPosZ + (par2EntityPlayer.posZ - par2EntityPlayer.prevPosZ) * (double)f;
637        Vec3 vec3 = par1World.getWorldVec3Pool().getVecFromPool(d0, d1, d2);
638        float f3 = MathHelper.cos(-f2 * 0.017453292F - (float)Math.PI);
639        float f4 = MathHelper.sin(-f2 * 0.017453292F - (float)Math.PI);
640        float f5 = -MathHelper.cos(-f1 * 0.017453292F);
641        float f6 = MathHelper.sin(-f1 * 0.017453292F);
642        float f7 = f4 * f5;
643        float f8 = f3 * f5;
644        double d3 = 5.0D;
645        if (par2EntityPlayer instanceof EntityPlayerMP)
646        {
647            d3 = ((EntityPlayerMP)par2EntityPlayer).theItemInWorldManager.getBlockReachDistance();
648        }
649        Vec3 vec31 = vec3.addVector((double)f7 * d3, (double)f6 * d3, (double)f8 * d3);
650        return par1World.rayTraceBlocks_do_do(vec3, vec31, par3, !par3);
651    }
