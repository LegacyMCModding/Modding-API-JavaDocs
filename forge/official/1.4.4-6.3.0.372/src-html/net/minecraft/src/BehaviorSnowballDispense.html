001    package net.minecraft.src;
002    
003    import net.minecraft.server.MinecraftServer;
004    
005    public class BehaviorSnowballDispense extends BehaviorProjectileDispense
006    {
007        /** Instance of MinecraftServer. */
008        final MinecraftServer theServer;
009    
010        public BehaviorSnowballDispense(MinecraftServer par1)
011        {
012            this.theServer = par1;
013        }
014    
015        /**
016         * Return the projectile entity spawned by this dispense behavior.
017         */
018        protected IProjectile getProjectileEntity(World par1World, IPosition par2IPosition)
019        {
020            return new EntitySnowball(par1World, par2IPosition.getX(), par2IPosition.getY(), par2IPosition.getZ());
021        }
022    }
