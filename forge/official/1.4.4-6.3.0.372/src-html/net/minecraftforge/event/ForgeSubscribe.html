001    package net.minecraftforge.event;
002    
003    import java.lang.annotation.Retention;
004    import java.lang.annotation.Target;
005    
006    import static java.lang.annotation.RetentionPolicy.*;
007    import static java.lang.annotation.ElementType.*;
008    
009    @Retention(value = RUNTIME)
010    @Target(value = METHOD)
011    public @interface ForgeSubscribe
012    {
013        public EventPriority priority() default EventPriority.NORMAL;
014        public boolean receiveCanceled() default false;
015    }
