001    package net.minecraft.src;
002    
003    import java.util.logging.ConsoleHandler;
004    import java.util.logging.FileHandler;
005    import java.util.logging.Level;
006    import java.util.logging.Logger;
007    
008    public class ConsoleLogManager
009    {
010        /** Log manager of Minecraft. */
011        public static Logger loggerLogManager = Logger.getLogger("Minecraft");
012    
013        /**
014         * Initialises the console logger.
015         */
016        public static void init()
017        {
018            ConsoleLogFormatter var0 = new ConsoleLogFormatter();
019            loggerLogManager.setUseParentHandlers(false);
020            ConsoleHandler var1 = new ConsoleHandler();
021            var1.setFormatter(var0);
022            loggerLogManager.addHandler(var1);
023    
024            try
025            {
026                FileHandler var2 = new FileHandler("server.log", true);
027                var2.setFormatter(var0);
028                loggerLogManager.addHandler(var2);
029            }
030            catch (Exception var3)
031            {
032                loggerLogManager.log(Level.WARNING, "Failed to log to server.log", var3);
033            }
034        }
035    }
