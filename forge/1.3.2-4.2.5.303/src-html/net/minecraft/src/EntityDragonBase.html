010            super(par1World);

022    
023        /**
024         * Called when the entity is attacked.
025         */
026        public boolean attackEntityFrom(DamageSource par1DamageSource, int par2)
027        {
028            return false;
029        }

035        {
036            return super.attackEntityFrom(par1DamageSource, par2);
